⚙ Advanced / one-time setup

Database lock — run this once in Supabase → SQL Editor so ONLY logged-in users can read or change your data:

-- 1) Table + realtime (safe to re-run; skips if it already exists)
create table if not exists scoopies_state (
  id text primary key default 'main',
  data jsonb not null default '{}'::jsonb,
  updated_at timestamptz not null default now()
);
insert into scoopies_state (id, data) values ('main','{}'::jsonb) on conflict (id) do nothing;
alter table scoopies_state enable row level security;
do $$ begin
  alter publication supabase_realtime add table scoopies_state;
exception when duplicate_object then null; end $$;

-- 2) Lock it down: remove the old "anyone" rules, allow ONLY logged-in users
drop policy if exists "scoopies read"   on scoopies_state;
drop policy if exists "scoopies insert" on scoopies_state;
drop policy if exists "scoopies update" on scoopies_state;
create policy "auth read"   on scoopies_state for select to authenticated using (true);
create policy "auth insert" on scoopies_state for insert to authenticated with check (true);
create policy "auth update" on scoopies_state for update to authenticated using (true) with check (true);

Setup order: in Supabase → Authentication turn OFF "Confirm email" → create your accounts on the login screen → run the SQL above → turn OFF new signups so no one else can register.
